#### General
````
ESC                     # stop ongoing operation
TAB                     # toggles edit/object mode
Z                       # toggles solid/wireframe display
CTRL + Z                # undo
SHIFT + CTRL + Z        # redo
CTRL + N                # open new default file
CTRL + O                # open file
CTRL + S                # save
SHIFT + CTRL + S        # save as
F12                     # render camera view
CTRL + U                # save as default
MOUSE 2                 # select
````

#### Navigation
````
CTRL + ALT + Q          # toggle quad view
CTRL + UP               # toggle full screen
NUMPAD 1                # front view
NUMPAD 3                # right side view
NUMPAD 7                # top view
CTRL + NUMPAD 1         # back view
CTRL + NUMPAD 3         # left side view
CTRL + NUMPAD 7         # bottom view
N                       # toggle 3D editor properties panel
T                       # toggle 3D editor tools panel
NUMPAD 0                # camera view
NUMPAD 5                # toggle perspective/orthographic
CTRL + ALT + NUMPAD 0   # align camera to view
J                       # cycle render slots
````

#### Editing
````
X                       # invoke X-axis
Y                       # invoke Y-axis
Z                       # invoke Z-axis
A                       # toggle select all/select none
B                       # box select
C                       # circle select
C + MOUSE 1             # circle mouse-over select
L                       # select linked
CTRL + I                # inverse selection
G                       # grab (move)
R                       # rotate
S                       # scale
X                       # delete
CTRL + TAB              # sub-object selection menu
ALT + MOUSE 2           # loop select
ALT + M                 # merge vertices
CTRL + ALT + MOUSE 2    # face loop select
CTRL + V                # vertex menu
CTRL + E                # edge menu
CTRL + F                # face menu
SPACE                   # search menu
CTRL + R                # loop cut and slide
K + MOUSE 1             # knife cut exact
SHIFT + K + MOUSE 1     # knife cut midpoints
V                       # rip tool
SHIFT + S               # snap menu
ALT + S                 # clear scale
ALT + R                 # clear rotation
ALT + G                 # clear location
O                       # proportional editing
F                       # make new face/edge
CTRL + TAB              # vertex/edge/face select
ALT + F                 # fill
SHIFT + ALT + F         # beauty fill
````
#### Modeling
````
SHIFT + A               # add object
E                       # extrude
CTRL + MOUSE 1          # extrude (here)
SHIFT + E               # extrude edges
SHIFT + D               # duplicate
ALT + D                 # linked duplicate
CTRL + P                # parent
ALT + P                 # clear parent
CTRL + J                # join objects
P                       # separate
ALT + C                 # convert
CTRL + H                # hooks
M                       # move to Layer
H                       # hide
SHIFT + CTRL + ALT + C  # set origin
SHIFT + C               # reset 3D cursor to 0,0,0
CTRL + SPACE            # turn 3D widget on/off
````
#### Curves
````
H                       # toggle free/aligned handles
V                       # vector handles
SHIFT + H               # auto handles
CTRL + MOUSE 1          # add control point
W                       # subdivide curve
````
#### Sculpting
````
F                       # change brush size
SHIFT + F               # change brush strength
CTRL + F                # rotate brush texture
````
